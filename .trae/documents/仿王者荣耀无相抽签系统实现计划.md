# 抽奖功能实现计划

## 一、技术选型

| 类别   | 技术/库                  | 理由                  |
| ---- | --------------------- | ------------------- |
| 路由管理 | Vue Router            | 实现主页面和抽奖页面的切换       |
| 状态管理 | Pinia                 | 复用现有状态管理系统，添加抽奖相关状态 |
| UI组件 | Vue 3 Composition API | 与现有项目技术栈保持一致        |

### 2. 抽奖状态管理

**创建lotteryStore.js**：

```javascript
// src/stores/lotteryStore.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useLotteryStore = defineStore('lottery', () => {
  // 状态
  const points = ref(0)           // 无相积分
  const stones = ref(0)           // 祈愿石数量
  const drawCount = ref(0)        // 累计抽奖次数
  const isExchanged = ref(false)  // 是否已兑换无相皮肤
  const firstDraw = ref(true)     // 是否首次单抽
  const history = ref([])         // 抽奖历史
  const bonuses = ref({
    5: false,
    10: false,
    25: false,
    45: false
  })

  // 奖励池配置
  const rewardPool = [
    { type: 'limited_skin', name: '限定皮肤礼（含无相）', probability: 0.4, points: 300 },
    { type: 'legendary_skin', name: '随机传说皮肤', probability: 0.5, points: 200 },
    { type: 'permanent_skin', name: '随机永久皮肤', probability: 13.5, points: 50 },
    { type: 'points_high', name: '无相积分', probability: 58.5, minPoints: 8, maxPoints: 68 },
    { type: 'points_low', name: '无相积分', probability: 27.1, minPoints: 1, maxPoints: 7 }
  ]

  // 计算属性
  const remainingGuarantee = computed(() => Math.max(0, 40 - drawCount.value))
  const canExchange = computed(() => points.value >= 998 && !isExchanged.value)

  // 方法
  function drawReward() {
    // 概率抽奖逻辑
    const random = Math.random() * 100
    let cumulativeProbability = 0

    for (const reward of rewardPool) {
      cumulativeProbability += reward.probability
      if (random <= cumulativeProbability) {
        if (reward.type === 'points_high' || reward.type === 'points_low') {
          // 随机积分值
          const pointsValue = Math.floor(Math.random() * (reward.maxPoints - reward.minPoints + 1)) + reward.minPoints
          return { ...reward, value: pointsValue }
        }
        return { ...reward, value: 1 }
      }
    }
  }

  function singleDraw() {
    // 单抽逻辑
    let cost = firstDraw.value ? 10 : 100
    if (stones.value < cost) return { success: false, message: '祈愿石不足' }

    stones.value -= cost
    firstDraw.value = false
    drawCount.value++

    const reward = drawReward()
    if (reward.type.includes('points')) {
      points.value += reward.value
    } else {
      // 处理皮肤奖励（这里简化处理，直接转换为积分）
      points.value += reward.points
    }

    // 检查额外奖励
    checkBonuses()
    // 检查次数保底
    checkGuarantee()
    // 保存历史记录
    saveHistory('single', [reward])

    return { success: true, reward }
  }

  function tenDraws() {
    // 10连抽逻辑
    const cost = 1000
    if (stones.value < cost) return { success: false, message: '祈愿石不足' }

    stones.value -= cost
    const rewards = []

    for (let i = 0; i < 10; i++) {
      drawCount.value++
      const reward = drawReward()
      if (reward.type.includes('points')) {
        points.value += reward.value
      } else {
        points.value += reward.points
      }
      rewards.push(reward)
    }

    // 检查额外奖励
    checkBonuses()
    // 检查次数保底
    checkGuarantee()
    // 保存历史记录
    saveHistory('ten', rewards)

    return { success: true, rewards }
  }

  function checkGuarantee() {
    // 次数保底检查
    if (drawCount.value >= 40) {
      // 强制发放永久皮肤（转换为积分）
      points.value += 50
      drawCount.value = 0
    }
  }

  function checkBonuses() {
    // 额外奖励检查
    if (drawCount.value >= 5 && !bonuses.value[5]) {
      points.value += 10
      bonuses.value[5] = true
    }
    if (drawCount.value >= 10 && !bonuses.value[10]) {
      points.value += 20
      bonuses.value[10] = true
    }
    if (drawCount.value >= 25 && !bonuses.value[25]) {
      points.value += 50
      bonuses.value[25] = true
    }
    if (drawCount.value >= 45 && !bonuses.value[45]) {
      points.value += 100
      bonuses.value[45] = true
    }
  }

  function exchangeSkin() {
    // 积分兑换逻辑
    if (canExchange.value) {
      points.value -= 998
      isExchanged.value = true
      return true
    }
    return false
  }

  function saveHistory(type, rewards) {
    // 保存抽奖历史
    history.value.push({
      id: Date.now().toString() + Math.random().toString(36).substr(2, 5),
      timestamp: new Date().toLocaleString(),
      type,
      rewards
    })
  }

  function resetData() {
    // 重置所有数据
    points.value = 0
    stones.value = 0
    drawCount.value = 0
    isExchanged.value = false
    firstDraw.value = true
    history.value = []
    bonuses.value = {
      5: false,
      10: false,
      25: false,
      45: false
    }
  }

  return {
    // 状态
    points,
    stones,
    drawCount,
    isExchanged,
    firstDraw,
    history,
    bonuses,
    rewardPool,
    // 计算属性
    remainingGuarantee,
    canExchange,
    // 方法
    drawReward,
    singleDraw,
    tenDraws,
    exchangeSkin,
    resetData
  }
}, {
  persist: {
    paths: ['points', 'stones', 'drawCount', 'isExchanged', 'firstDraw', 'history', 'bonuses']
  }
})
```

<script setup>
import { ref } from 'vue'
import { useLotteryStore } from '../stores/lotteryStore'

const lotteryStore = useLotteryStore()
const lastResult = ref(null)
const addStones = ref(0)

function handleSingleDraw() {
  const result = lotteryStore.singleDraw()
  if (result.success) {
    lastResult.value = { rewards: [result.reward] }
  } else {
    alert(result.message)
  }
}

function handleTenDraws() {
  const result = lotteryStore.tenDraws()
  if (result.success) {
    lastResult.value = { rewards: result.rewards }
  } else {
    alert(result.message)
  }
}

function handleExchange() {
  const success = lotteryStore.exchangeSkin()
  if (success) {
    alert('兑换成功！')
  }
}

function handleReset() {
  if (confirm('确定要重置所有数据吗？')) {
    lotteryStore.resetData()
    lastResult.value = null
  }
}

function handleAddStones() {
  if (addStones.value > 0) {
    lotteryStore.stones += addStones.value
    addStones.value = 0
  }
}

function clearHistory() {
  if (confirm('确定要清空抽奖历史吗？')) {
    lotteryStore.history = []
  }
}
</script>
