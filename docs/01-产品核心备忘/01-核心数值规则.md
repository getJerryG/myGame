## 一、四大核心数值计算公式

### 1. 资金 (money)

```typescript
// 每日资金计算
const dailyMoney = yesterdayMoney + dailyIncome - dailyCost + rewards;

// 每日收入 (dailyIncome)
const dailyIncome = heroIncome + skinIncome + activityIncome * popularityMultiplier;

// 英雄每日收入
const heroIncome = onlineHeroCount * randomRange(100, 300) * winRateMultiplier;
// winRateMultiplier: 胜率45%-55%对应0.8-1.2倍

// 皮肤每日收入
const skinIncome = Σ(skin.dailySales * skin.price);
// 皮肤日销量见下表

// 活动收入
const activityIncome = {
  '轻度': 500,
  '中度': 1000,
  '重磅': 2000
}[activityType];

// 热度加成系数
const popularityMultiplier = popularity / 100; // 热度50=0.5倍, 热度200=2倍
```

### 2. 声望 (reputation)

```typescript
// 声望计算
const reputation = initialReputation + totalGain - totalLoss;

// 升级奖励（资金和声望）
// 详细奖励配置请参阅：02-策划等级体系详细说明.md
import { promotionRewards } from './02-策划等级体系详细说明';

// 声望提升
const gainSources = {
  '升级奖励': (from, to) => promotionRewards[`${from}→${to}`]?.reputation || 0,
  '联动成功': randomRange(20, 50),
  '活动成功': 10,
  '累计流水': Math.floor(totalMoney / 100000) * 5
};

// 资金奖励（在升级时调用）
const getPromotionMoneyReward = (from, to) => {
  return promotionRewards[`${from}→${to}`]?.money || 0;
};

// 声望下降
const lossSources = {
  '联动失败': randomRange(20, 50),
  '事件处理失败': randomRange(10, 30)
};
```

### 3. 热度 (popularity)

```typescript
// 热度计算
const popularity = yesterdayPopularity + gain - loss;

// 热度提升
const popularityGain = {
  '新英雄上线': 5,
  '新皮肤上线': 3,
  '传说/限定皮肤': 8,
  '联动成功': randomRange(10, 30),
  '活动开启': { '轻度': 2, '中度': 5, '重磅': 10 },
  '事件处理成功': 2
};

// 热度下降
const popularityLoss = {
  '事件处理失败': randomRange(5, 10),
  '英雄胜率异常': 3, // 胜率过低/过高
  '皮肤口碑差': 5,
  '长时间无更新': 1 // 每天
};
```

### 4. 口碑 (wordOfMouth)

```typescript
// 口碑计算
const wordOfMouth = yesterdayWOM + gain - loss;
```

## 二、皮肤销量计算公式

### 皮肤日销量基础值

| 品质 | 日销量范围 | 加成系数 |
|------|-----------|---------|
| 伴生 | 50-100 | 1.0 |
| 勇者 | 100-200 | 1.2 |
| 史诗 | 200-400 | 1.5 |
| 传说 | 400-800 | 2.0 |
| 限定/典藏 | 800-1500 | 3.0 |

```typescript
// 皮肤日销量计算
const dailySales = baseSales * qualityMultiplier * popularityMultiplier * levelBonus;

// 品质系数
const qualityMultiplier = {
  'companion': 1.0,
  'brave': 1.2,
  'epic': 1.5,
  'legend': 2.0,
  'limited': 3.0,
  'mythic': 3.0
}[skinQuality];

// 职级销量加成
const levelBonus = {
  '见习': 1.0,
  '初级': 1.0,
  '中级': 1.0,
  '高级': 1.0,
  '资深': 1.05,
  '专家': 1.10,
  '经理': 1.15,
  '总监': 1.25
}[level];
```

## 六、快速参考

### 常用随机数函数

```typescript
// 生成指定范围的随机整数
const randomRange = (min: number, max: number): number => {
  return Math.floor(Math.random() * (max - min + 1)) + min;
};

// 生成指定范围的随机数（保留2位小数）
const randomRangeFloat = (min: number, max: number): number => {
  return Number((Math.random() * (max - min) + min).toFixed(2));
};
```

### 数值范围常量

```typescript
const VALUE_RANGES = {
  money: { min: 0, max: Infinity },
  reputation: { min: 0, max: 5000 },
  popularity: { min: 0, max: 200 },
  wordOfMouth: { min: -100, max: 100 }
};

// 数值边界检查
const clampValue = (value: number, min: number, max: number): number => {
  return Math.max(min, Math.min(max, value));
};
```
