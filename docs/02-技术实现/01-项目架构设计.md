## 三、状态管理方案（Pinia Store拆分）

### 3.1 Store划分

```typescript
// stores/gameStore.ts - 游戏核心状态
export const useGameStore = defineStore("game", {
  state: () => ({
    // 时间系统
    currentDate: { year: 1, month: 1, day: 1 },
    isPlayerTurn: true,

    // 核心数值
    money: 10000,
    reputation: 0,
    popularity: 50,
    wordOfMouth: 100,
    totalMoney: 0,

    // 统计
    heroCount: 0,
    skinCount: 0,

    // 策划等级
    plannerLevel: "见习",
    plannerSubLevel: "III",
    plannerExp: 0,

    // 进行中的项目
    ongoingProjects: [],

    // 已上线内容
    onlineHeroes: [],
    onlineSkins: [],
    activeEvents: [],
  }),

  getters: {
    // 计算属性
    expProgressPercent: (state) => {
      return Math.floor(
        (state.plannerExp / getMaxExp(state.plannerLevel)) * 100,
      );
    },

    canPromote: (state) => {
      return checkPromotionRequirements(state);
    },
  },

  actions: {
    // 每日结算
    dailySettlement() {
      this.calculateIncome();
      this.updateProjects();
      this.checkRandomEvents();
      this.checkPromotion();
    },

    // 推进时间
    nextDay() {
      this.currentDate.day++;
      if (this.currentDate.day > 30) {
        this.currentDate.day = 1;
        this.currentDate.month++;
        if (this.currentDate.month > 12) {
          this.currentDate.month = 1;
          this.currentDate.year++;
        }
      }
      this.isPlayerTurn = false;
      this.dailySettlement();
      this.isPlayerTurn = true;
    },
  },
});

// stores/simulationStore.ts - 模拟系统状态
export const useSimulationStore = defineStore("simulation", {
  state: () => ({
    // 模拟数据
    simulationData: null,
    isSimulating: false,
    simulationSpeed: 1,
  }),

  actions: {
    startSimulation() {
      this.isSimulating = true;
    },

    stopSimulation() {
      this.isSimulating = false;
    },

    updateSimulationData(data: any) {
      this.simulationData = data;
    },
  },
});

// stores/saveStore.ts - 存档状态
export const useSaveStore = defineStore("save", {
  state: () => ({
    saves: [],
    currentSaveId: null,
    autoSaveInterval: null,
  }),

  actions: {
    // 创建存档
    createSave() {
      const saveData = this.serializeGameData();
      const saveId = Date.now().toString();
      localStorage.setItem(`game_save_${saveId}`, JSON.stringify(saveData));
      this.saves.push({
        id: saveId,
        date: new Date(),
        preview: this.generatePreview(saveData),
      });
    },

    // 加载存档
    loadSave(saveId: string) {
      const saveData = localStorage.getItem(`game_save_${saveId}`);
      if (saveData) {
        this.deserializeGameData(JSON.parse(saveData));
        this.currentSaveId = saveId;
        return true;
      }
      return false;
    },

    // 自动存档
    startAutoSave() {
      this.autoSaveInterval = setInterval(
        () => {
          this.createSave();
        },
        5 * 60 * 1000,
      ); // 5分钟
    },
  },
});
```

### 4.2 模块职责


| 模块 | 职责 | 关键功能 |
|------|------|---------|
| Time System | 时间管理 | 日期推进、每日结算触发 |
| Value System | 数值计算 | 资金/声望/热度/口碑计算 |
| Archive System | 数据持久化 | 存档/读档/自动保存 |
| Hero System | 英雄管理 | 研发/上线/数据统计 |
| Skin System | 皮肤管理 | 研发/销量计算/统计 |
| Activity System | 活动管理 | 开启/收益计算/结束 |
| Event System | 事件处理 | 随机触发/选择处理/数值影响 |
| Level System | 职级系统 | 升级检查/权限解锁/奖励 |
