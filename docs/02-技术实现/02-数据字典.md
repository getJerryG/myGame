## 一、核心变量定义

### 1.1 游戏状态变量


| 字段名 | 类型 | 默认值 | 说明 |
|--------|------|--------|------|
| money | number | 10000 | 当前资金 |
| reputation | number | 0 | 声望值 |
| popularity | number | 50 | 游戏热度 |
| wordOfMouth | number | 100 | 口碑值 |
| totalMoney | number | 0 | 累计总流水 |
| heroCount | number | 0 | 已上线英雄总数 |
| skinCount | number | 0 | 已上线皮肤总数 |

```typescript
// 游戏核心状态接口
interface GameState {
  money: number; // 当前资金
  reputation: number; // 声望值
  popularity: number; // 游戏热度
  wordOfMouth: number; // 口碑值
  totalMoney: number; // 累计总流水
  heroCount: number; // 已上线英雄总数
  skinCount: number; // 已上线皮肤总数
}

// 初始状态
const initialGameState: GameState = {
  money: 10000,
  reputation: 0,
  popularity: 50,
  wordOfMouth: 100,
  totalMoney: 0,
  heroCount: 0,
  skinCount: 0,
};
```

### 1.2 时间系统变量


| 字段名 | 类型 | 默认值 | 说明 |
|--------|------|--------|------|
| year | number | 1 | 当前年份 |
| month | number | 1 | 当前月份 (1-12) |
| day | number | 1 | 当前日期 (1-30) |
| isPlayerTurn | boolean | true | 是否玩家决策阶段 |

```typescript
// 日期接口
interface GameDate {
  year: number; // 1 ~ ∞
  month: number; // 1 ~ 12
  day: number; // 1 ~ 30
}

// 时间状态
interface TimeState {
  currentDate: GameDate;
  isPlayerTurn: boolean;
}

// 初始时间
const initialTimeState: TimeState = {
  currentDate: { year: 1, month: 1, day: 1 },
  isPlayerTurn: true,
};
```

### 1.3 策划等级变量


| 字段名 | 类型 | 默认值 | 说明 |
|--------|------|--------|------|
| plannerLevel | string | '见习' | 策划职级大类 |
| plannerSubLevel | string | 'III' | 策划子等级 |
| plannerExp | number | 0 | 当前经验值 |

```typescript
// 策划等级状态
interface PlannerLevelState {

// 初始等级
const initialPlannerLevel: PlannerLevelState = {
```

## 三、本地存储Key清单

### 3.1 存档相关

```typescript
// 存档Key前缀
const STORAGE_KEY_PREFIX = "game_";

// 具体Key
const STORAGE_KEYS = {
  // 存档列表
  SAVE_LIST: `${STORAGE_KEY_PREFIX}save_list`,

  // 单个存档（动态生成）
  SAVE_DATA: (saveId: string) => `${STORAGE_KEY_PREFIX}save_${saveId}`,

  // 当前存档ID
  CURRENT_SAVE_ID: `${STORAGE_KEY_PREFIX}current_save_id`,

  // 游戏设置
  SETTINGS: `${STORAGE_KEY_PREFIX}settings`,

  // 用户数据
  USER_DATA: `${STORAGE_KEY_PREFIX}user_data`,

  // 统计数据
  STATISTICS: `${STORAGE_KEY_PREFIX}statistics`,
};

// 使用示例
localStorage.setItem(
  STORAGE_KEYS.SAVE_DATA("123456"),
  JSON.stringify(gameData),
);
const saveData = JSON.parse(
  localStorage.getItem(STORAGE_KEYS.SAVE_DATA("123456")) || "{}",
);
```

```typescript
// 存档数据结构
```

## 四、接口类型定义

### 4.1 英雄相关

```typescript
// 英雄
interface Hero {
  id: string; // 英雄ID
  name: string; // 英雄名称
  class: HeroClass; // 职业
  style: HeroStyle; // 流派
  status: ProjectStatus; // 状态

  // 研发信息
  researchProgress: number; // 研发进度 (0-100)
  researchDays: number; // 已研发天数
  totalResearchDays: number; // 总研发天数
  researchCost: number; // 研发成本

  // 上线数据
  onlineDate?: GameDate; // 上线日期
  winRate: number; // 胜率
  pickRate: number; // 登场率
  banRate: number; // 禁用率
  isHit: boolean; // 是否爆款
}

// 英雄研发项目
interface HeroProject {
  id: string;
  heroId: string;
  startDate: GameDate;
  progress: number;
  status: ProjectStatus;
}
```

```typescript
// 皮肤
interface Skin {
  id: string; // 皮肤ID
```

### 4.3 事件相关

```typescript
// 随机事件
interface RandomEvent {
  id: string; // 事件ID
  title: string; // 事件标题
  description: string; // 事件描述
  type: "positive" | "negative" | "neutral"; // 事件类型
  probability: number; // 触发概率 (0-1)

  // 处理选项
  options: EventOption[];
}

// 事件选项
interface EventOption {
  id: string; // 选项ID
  text: string; // 选项文本
  cost: number; // 消耗资金

  // 数值影响
  effects: {
    money?: number; // 资金变化
    reputation?: number; // 声望变化
    popularity?: number; // 热度变化
    wordOfMouth?: number; // 口碑变化
  };
}

// 进行中的事件
interface ActiveEvent {
  eventId: string;
  startDate: GameDate;
  selectedOption: string;
  isResolved: boolean;
}
```
